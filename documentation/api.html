var vDOM = require('./vDOM.js');
module.exports = function(nodes) {
    var api = {
        append: function(arg) {
            switch (typeof arg) {
                case "string":
                        vDOM.addChildFromHtml(this, arg);
                        return this;
            }
        },
        remove: function() {
            vDOM.removeNode(this);
            return this;
        },
        addClass: function(classes) {
            vDOM.addClasses(this, classes);
            return this;
        },
        html: function(html) {
            vDOM.setHTML(this, html);
            return this;
        }
    }
    return Object.assign(nodes, api);
}
